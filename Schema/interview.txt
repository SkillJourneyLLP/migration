create table public.interview (
  id uuid not null default gen_random_uuid (),
  user_id uuid not null,
  position text not null,
  job_description text null,
  duration integer not null default 30,
  evaluation_criteria text null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  start_date date null,
  start_time time without time zone null,
  end_date date null,
  end_time time without time zone null,
  status text null default 'Scheduled'::text,
  opening_statement text null,
  report_gen_criteria text null,
  application_link text null,
  faq text null,
  organization_id uuid null,
  "noteForUser" text null,
  deactive boolean null,
  is_deleted boolean null default false,
  is_archived boolean null default false,
  language text null,
  psychometric boolean null,
  phone_verification text null,
  constraint interview_pkey primary key (id),
  constraint interview_organization_id_fkey foreign KEY (organization_id) references organizations (id),
  constraint interview_user_id_fkey foreign KEY (user_id) references users (id)
) TABLESPACE pg_default;

create trigger update_interview_updated_at BEFORE
update on interview for EACH row
execute FUNCTION update_updated_at_column ();
