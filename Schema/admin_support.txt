create table public.admin_support (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone null,
  support_type text null,
  support_description text null,
  status text null,
  note_for_the_admin text null,
  note text null,
  link text null,
  org_id uuid null,
  created_by text null,
  resolved_at timestamp with time zone null,
  constraint admin_support_pkey primary key (id),
  constraint admin_support_org_id_fkey foreign KEY (org_id) references organizations (id)
) TABLESPACE pg_default;
