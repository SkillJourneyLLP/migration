create table public.eval_criteria (
  id uuid not null default gen_random_uuid (),
  interview_id uuid not null,
  criteria_name text not null,
  weightage integer not null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  constraint eval_criteria_pkey primary key (id),
  constraint eval_criteria_weightage_check check (
    (
      (weightage >= 0)
      and (weightage <= 100)
    )
  )
) TABLESPACE pg_default;

create trigger update_eval_criteria_updated_at BEFORE
update on eval_criteria for EACH row
execute FUNCTION update_updated_at_column ();
