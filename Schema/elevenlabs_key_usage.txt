create table public.elevenlabs_key_usage (
  key_index integer not null,
  current_active_sessions integer not null default 0,
  max_concurrent_sessions integer not null default 5,
  last_used_at timestamp with time zone null default now(),
  is_active boolean not null default true,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  available_credits integer null,
  last_credit_check timestamp with time zone null,
  credit_threshold integer null default 4000,
  constraint elevenlabs_key_usage_pkey primary key (key_index),
  constraint elevenlabs_key_usage_current_active_sessions_check check ((current_active_sessions >= 0)),
  constraint elevenlabs_key_usage_key_index_check check (
    (
      (key_index >= 1)
      and (key_index <= 6)
    )
  ),
  constraint elevenlabs_key_usage_max_concurrent_sessions_check check ((max_concurrent_sessions > 0))
) TABLESPACE pg_default;

create index IF not exists idx_elevenlabs_key_usage_active on public.elevenlabs_key_usage using btree (is_active) TABLESPACE pg_default;

create trigger update_elevenlabs_key_usage_updated_at BEFORE
update on elevenlabs_key_usage for EACH row
execute FUNCTION update_updated_at_column ();
