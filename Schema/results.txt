create table public.results (
  id uuid not null default gen_random_uuid (),
  candidate_id uuid not null,
  interview_id uuid not null,
  name text not null,
  email text null,
  status text not null default 'pending'::text,
  score text null,
  recommendation text null,
  report_link text null,
  created_at timestamp with time zone not null default now(),
  updated_at timestamp with time zone not null default now(),
  phone text null,
  resume_summary text null,
  location text null,
  current_ctc text null,
  expected_ctc text null,
  gender text null,
  availability text null,
  work_preference text null,
  is_referred boolean null default false,
  referer_name text null,
  referer_email text null,
  ai_proctoring text null,
  report_generated boolean null,
  duration text null,
  attempted_at timestamp with time zone null,
  proctoring text null,
  constraint results_pkey primary key (id)
) TABLESPACE pg_default;
